## Installation

Install rc-charts from npm:

```
npm install rc-charts || yarn add rc-charts
```

## Usage

```
import React from 'react';
import { Pie } form 'rc-charts';

const App: React.FC = () => {
  return (
    <div>
      <Pie
        data={[
          { x: 'chrome', y: 20 },
          { x: 'IE', y: 20 },
          { x: 'Firefox', y: 20 }
        ]}
        tooltip={true}
        subTitle="总预警数"
        total={100}
      />
    </div>
  );
}

export default App
```
